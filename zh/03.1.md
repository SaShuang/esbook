# 3.1 统计ES开发

## @挨踢熊

	select rtid,
		rtno 编号编号,
		rtname 模板名称,
		classid,
		creuname 创建人,
		cretime 创建时间,
		updtime 更新时间,
		ver 
	from es_tmp
	order by ver desc
	
	select sum(ver) 总编辑次数,
		count(rtid) 模板数量,
		max(ver) 最高编辑次数,
		sum(ver)/count(rtid) 平均每个模板编辑次数 
	from es_tmp
